```python
import pytesseract
from PIL import Image
import cv2
import numpy as np

def ocr_image(image_path, lang='chi_sim+eng', preprocess=True):
    """
    иҜҶеҲ«еӣҫзүҮдёӯзҡ„ж–Үеӯ—
    
    Args:
        image_path: еӣҫзүҮи·Ҝеҫ„
        lang: иҜҶеҲ«иҜӯиЁҖпјҲchi_sim:з®ҖдҪ“дёӯж–Ү, eng:иӢұж–Үпјү
        preprocess: жҳҜеҗҰиҝӣиЎҢеӣҫеғҸйў„еӨ„зҗҶ
    
    Returns:
        иҜҶеҲ«зҡ„ж–Үеӯ—еҶ…е®№
    """
    # иҜ»еҸ–еӣҫзүҮ
    image = cv2.imread(image_path)
    
    if preprocess:
        # еӣҫеғҸйў„еӨ„зҗҶ
        image = preprocess_image(image)
    
    # иҪ¬жҚўдёә PIL Image
    pil_image = Image.fromarray(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))
    
    # OCR иҜҶеҲ«
    text = pytesseract.image_to_string(pil_image, lang=lang)
    
    return text.strip()

def preprocess_image(image):
    """
    еӣҫеғҸйў„еӨ„зҗҶпјҢжҸҗй«ҳиҜҶеҲ«еҮҶзЎ®зҺҮ
    
    Args:
        image: OpenCV еӣҫеғҸеҜ№иұЎ
    """
    # иҪ¬дёәзҒ°еәҰеӣҫ
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    
    # еҺ»еҷӘ
    denoised = cv2.fastNlMeansDenoising(gray)
    
    # дәҢеҖјеҢ–
    _, binary = cv2.threshold(denoised, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
    
    # еҖҫж–ңж ЎжӯЈ
    corrected = deskew(binary)
    
    return corrected

def deskew(image):
    """еҖҫж–ңж ЎжӯЈ"""
    coords = np.column_stack(np.where(image > 0))
    angle = cv2.minAreaRect(coords)[-1]
    
    if angle < -45:
        angle = -(90 + angle)
    else:
        angle = -angle
    
    (h, w) = image.shape[:2]
    center = (w // 2, h // 2)
    M = cv2.getRotationMatrix2D(center, angle, 1.0)
    rotated = cv2.warpAffine(image, M, (w, h),
                             flags=cv2.INTER_CUBIC,
                             borderMode=cv2.BORDER_REPLICATE)
    
    return rotated
```

**дҪҝз”Ё PaddleOCRпјҲжҺЁиҚҗпјүпјҡ**

```python
from paddleocr import PaddleOCR
import cv2

# еҲқе§ӢеҢ– OCR еј•ж“Һ
ocr_engine = PaddleOCR(
    use_angle_cls=True,  # дҪҝз”Ёж–№еҗ‘еҲҶзұ»еҷЁ
    lang='ch',           # дёӯж–Ү
    use_gpu=False        # CPU иҝҗиЎҢ
)

def paddle_ocr(image_path):
    """
    дҪҝз”Ё PaddleOCR иҜҶеҲ«еӣҫзүҮ
    
    Args:
        image_path: еӣҫзүҮи·Ҝеҫ„
    
    Returns:
        иҜҶеҲ«з»“жһңеҲ—иЎЁпјҢжҜҸдёӘе…ғзҙ еҢ…еҗ«ж–Үеӯ—гҖҒдҪҚзҪ®е’ҢзҪ®дҝЎеәҰ
    """
    result = ocr_engine.ocr(image_path, cls=True)
    
    recognized_text = []
    for line in result[0]:
        if line:
            bbox = line[0]      # ж–Үеӯ—жЎҶдҪҚзҪ®
            text = line[1][0]   # ж–Үеӯ—еҶ…е®№
            confidence = line[1][1]  # зҪ®дҝЎеәҰ
            
            recognized_text.append({
                'text': text,
                'bbox': bbox,
                'confidence': confidence
            })
    
    return recognized_text

def extract_text_only(image_path):
    """д»…жҸҗеҸ–ж–Үеӯ—еҶ…е®№"""
    result = paddle_ocr(image_path)
    return '\n'.join([item['text'] for item in result])
```

```python
import re

def recognize_invoice(image_path):
    """
    иҜҶеҲ«еҸ‘зҘЁдҝЎжҒҜ
    
    Args:
        image_path: еҸ‘зҘЁеӣҫзүҮи·Ҝеҫ„
    
    Returns:
        з»“жһ„еҢ–еҸ‘зҘЁдҝЎжҒҜ
    """
    # OCR иҜҶеҲ«
    text = extract_text_only(image_path)
    
    # жҸҗеҸ–е…ій”®дҝЎжҒҜ
    invoice_info = {
        'invoice_code': extract_pattern(text, r'еҸ‘зҘЁд»Јз Ғ[:пјҡ]\s*(\d+)'),
        'invoice_number': extract_pattern(text, r'еҸ‘зҘЁеҸ·з Ғ[:пјҡ]\s*(\d+)'),
        'date': extract_pattern(text, r'(\d{4}е№ҙ\d{1,2}жңҲ\d{1,2}ж—Ҙ|\d{4}-\d{2}-\d{2})'),
        'amount': extract_pattern(text, r'(?:д»·зЁҺеҗҲи®Ў|йҮ‘йўқ)[:пјҡ]?\s*[ВҘпҝҘ]?\s*([\d,\.]+)'),
        'seller': extract_pattern(text, r'й”Җе”®ж–№.*?еҗҚз§°[:пјҡ]\s*([^\n]+)'),
        'buyer': extract_pattern(text, r'иҙӯд№°ж–№.*?еҗҚз§°[:пјҡ]\s*([^\n]+)'),
        'tax_id': extract_pattern(text, r'зәізЁҺдәәиҜҶеҲ«еҸ·[:пјҡ]\s*([A-Z0-9]+)')
    }
    
    return invoice_info

def extract_pattern(text, pattern):
    """дҪҝз”ЁжӯЈеҲҷжҸҗеҸ–еҶ…е®№"""
    match = re.search(pattern, text, re.DOTALL)
    return match.group(1).strip() if match else None
```

**еҗҚзүҮиҜҶеҲ«пјҡ**

```python
def recognize_business_card(image_path):
    """
    иҜҶеҲ«еҗҚзүҮдҝЎжҒҜ
    
    Args:
        image_path: еҗҚзүҮеӣҫзүҮи·Ҝеҫ„
    
    Returns:
        з»“жһ„еҢ–еҗҚзүҮдҝЎжҒҜ
    """
    text = extract_text_only(image_path)
    lines = text.split('\n')
    
    card_info = {
        'name': None,
        'company': None,
        'title': None,
        'phone': None,
        'email': None,
        'address': None
    }
    
    for line in lines:
        line = line.strip()
        
        # иҜҶеҲ«жүӢжңәеҸ·
        phone_match = re.search(r'1[3-9]\d{9}', line)
        if phone_match:
            card_info['phone'] = phone_match.group()
            continue
        
        # иҜҶеҲ«йӮ®з®ұ
        email_match = re.search(r'[\w.-]+@[\w.-]+\.\w+', line)
        if email_match:
            card_info['email'] = email_match.group()
            continue
        
        # иҜҶеҲ«е…¬еҸёпјҲйҖҡеёёеҢ…еҗ«"е…¬еҸё"гҖҒ"йӣҶеӣў"зӯүпјү
        if any(keyword in line for keyword in ['е…¬еҸё', 'йӣҶеӣў', 'дјҒдёҡ', 'з§‘жҠҖ']):
            card_info['company'] = line
            continue
        
        # иҜҶеҲ«иҒҢдҪҚпјҲйҖҡеёёеҢ…еҗ«"з»ҸзҗҶ"гҖҒ"жҖ»зӣ‘"зӯүпјү
        if any(keyword in line for keyword in ['з»ҸзҗҶ', 'жҖ»зӣ‘', 'дё»з®Ў', 'е·ҘзЁӢеёҲ']):
            card_info['title'] = line
            continue
        
        # иҜҶеҲ«е§“еҗҚпјҲйҖҡеёёжҳҜзҹӯж–Үжң¬пјҢ2-4дёӘеӯ—з¬Ұпјү
        if 2 <= len(line) <= 4 and not any(c.isdigit() for c in line):
            if not card_info['name']:
                card_info['name'] = line
    
    return card_info
```

```python
def recognize_table(image_path):
    """
    иҜҶеҲ«еӣҫзүҮдёӯзҡ„иЎЁж ј
    
    Args:
        image_path: еӣҫзүҮи·Ҝеҫ„
    
    Returns:
        иЎЁж јж•°жҚ®пјҲдәҢз»ҙеҲ—иЎЁпјү
    """
    # дҪҝз”Ё PaddleOCR зҡ„иЎЁж јиҜҶеҲ«
    from paddleocr import PPStructure
    
    table_engine = PPStructure(
        show_log=False,
        layout=False,
        table=True
    )
    
    img = cv2.imread(image_path)
    result = table_engine(img)
    
    tables = []
    for line in result:
        if line['type'] == 'table':
            # жҸҗеҸ–иЎЁж ј HTML
            html = line['res']['html']
            # и§Јжһҗ HTML дёәдәҢз»ҙеҲ—иЎЁ
            table_data = parse_table_html(html)
            tables.append(table_data)
    
    return tables

def parse_table_html(html):
    """и§ЈжһҗиЎЁж ј HTML дёәдәҢз»ҙеҲ—иЎЁ"""
    from bs4 import BeautifulSoup
    
    soup = BeautifulSoup(html, 'html.parser')
    table = soup.find('table')
    
    data = []
    for row in table.find_all('tr'):
        row_data = []
        for cell in row.find_all(['td', 'th']):
            row_data.append(cell.get_text().strip())
        data.append(row_data)
    
    return data

def table_to_excel(table_data, output_path):
    """е°ҶиЎЁж јж•°жҚ®дҝқеӯҳдёә Excel"""
    import pandas as pd
    
    df = pd.DataFrame(table_data[1:], columns=table_data[0])
    df.to_excel(output_path, index=False)
    
    return output_path
```

```python
def format_ocr_results_as_text(results):
```

```python
class OCRSkill(Skill):
    name = "OCRжҷәиғҪиҜҶеҲ«"
    description = "иҜҶеҲ«еӣҫзүҮе’Ңж–ҮжЎЈдёӯзҡ„ж–Үеӯ—"
    
    def __init__(self):
        self.ocr = PaddleOCR(use_angle_cls=True, lang='ch', use_gpu=False)
    
    @Tool
    def recognize_text(self, image_path: str) -> str:
        """иҜҶеҲ«еӣҫзүҮдёӯзҡ„ж–Үеӯ—"""
        result = self.ocr.ocr(image_path, cls=True)
        texts = [line[1][0] for line in result[0] if line]
        return '\n'.join(texts)
    
    @Tool
    def recognize_invoice(self, image_path: str) -> dict:
        """иҜҶеҲ«еҸ‘зҘЁдҝЎжҒҜ"""
        text = self.recognize_text(image_path)
        # жҸҗеҸ–з»“жһ„еҢ–дҝЎжҒҜ
        return {
            'invoice_code': self._extract(text, r'еҸ‘зҘЁд»Јз Ғ[:пјҡ]\s*(\d+)'),
            'invoice_number': self._extract(text, r'еҸ‘зҘЁеҸ·з Ғ[:пјҡ]\s*(\d+)'),
            'amount': self._extract(text, r'д»·зЁҺеҗҲи®Ў.*?([\d,\.]+)'),
            'date': self._extract(text, r'(\d{4}е№ҙ\d{1,2}жңҲ\d{1,2}ж—Ҙ)')
        }
    
    @Tool
    def recognize_table(self, image_path: str, output: str) -> str:
        """иҜҶеҲ«еӣҫзүҮдёӯзҡ„иЎЁж је№¶еҜјеҮә Excel"""
        # иЎЁж јиҜҶеҲ«йҖ»иҫ‘
        # ...
        return f"иЎЁж је·ІеҜјеҮәеҲ° {output}"
    
    def _extract(self, text, pattern):
        import re
        match = re.search(pattern, text)
        return match.group(1) if match else None
```

### 5.1 жЎҲдҫӢдёҖпјҡеҸ‘зҘЁиҮӘеҠЁеҪ•е…Ҙ

```python
def batch_process_invoices(image_folder):
    """
    жү№йҮҸеӨ„зҗҶеҸ‘зҘЁ
    
    Args:
        image_folder: еҸ‘зҘЁеӣҫзүҮж–Үд»¶еӨ№
    """
    import os
    import pandas as pd
    
    results = []
    
    for filename in os.listdir(image_folder):
        if filename.lower().endswith(('.png', '.jpg', '.jpeg')):
            image_path = os.path.join(image_folder, filename)
            
            # иҜҶеҲ«еҸ‘зҘЁ
            invoice_info = recognize_invoice(image_path)
            invoice_info['filename'] = filename
            invoice_info['status'] = 'жҲҗеҠҹ' if invoice_info['invoice_code'] else 'йңҖдәәе·Ҙж ёеҜ№'
            
            results.append(invoice_info)
    
    # еҜјеҮәдёә Excel
    df = pd.DataFrame(results)
    output_path = os.path.join(image_folder, 'еҸ‘зҘЁиҜҶеҲ«з»“жһң.xlsx')
    df.to_excel(output_path, index=False)
    
    return output_path
```

```python
def generate_vcards(contacts, output_path):
    """з”ҹжҲҗ vCard ж–Үд»¶"""
    with open(output_path, 'w', encoding='utf-8') as f:
        for contact in contacts:
            f.write('BEGIN:VCARD\n')
            f.write('VERSION:3.0\n')
            if contact.get('name'):
                f.write(f"FN:{contact['name']}\n")
            if contact.get('phone'):
                f.write(f"TEL:{contact['phone']}\n")
            if contact.get('email'):
                f.write(f"EMAIL:{contact['email']}\n")
            if contact.get('company'):
                f.write(f"ORG:{contact['company']}\n")
            if contact.get('title'):
                f.write(f"TITLE:{contact['title']}\n")
            f.write('END:VCARD\n')
```
