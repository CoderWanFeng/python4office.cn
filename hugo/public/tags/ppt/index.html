        'price': '299元',

for p in products:
    filename = f"PPT_{p['name']}.pptx"
    
    # 创建PPT
    office.ppt.create(path=filename, title=p['name'])
    
    # 添加封面
    office.ppt.add(
        path=filename,
        title=p['name'],
        content=f"价格：{p['price']}"
    )
    
    # 添加介绍
    office.ppt.add(
        path=filename,
        title='课程介绍',
        content=p['desc']
    )
    
    # 添加适用人群
    office.ppt.add(
        path=filename,
        title='适用人群',
        content=p['target']
    )
    
    # 添加购买方式
    office.ppt.add(
        path=filename,
        title='购买方式',

    )
    
    print(f'已生成: {filename}')

print(f'共生成 {len(products)} 个PPT！')

2、中级：带图片的PPT

import office

def create_product_ppt(product, image_path):
    """创建带图片的产品PPT"""
    filename = f"PPT_{product['name']}.pptx"
    
    # 封面
    office.ppt.create(path=filename, title=product['name'])
    
    # 带图片的介绍页
    office.ppt.add(
        path=filename,
        title=product['name'],
        content=f"💰 价格：{product['price']}",
        image=image_path
    )
    
    # 详情页
    office.ppt.add(
        path=filename,
        title='课程详情',
        content=product['desc']
    )
    
    return filename

# 批量创建
