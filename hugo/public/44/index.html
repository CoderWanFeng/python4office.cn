1、什么是函数？

函数就是一段可以重复使用的代码块：

# 定义函数
def say_hello():
    print('Hello！')
    print('你好！')

# 调用函数
say_hello()

2、函数的参数

# 带参数函数
def greet(name):
    print(f'你好，{name}！')

greet('张三')
greet('李四')

# 带默认参数
def greet(name, age=18):
    print(f'你好，{name}！你{age}岁了。')

greet('张三')        # 使用默认年龄
greet('李四', 20)    # 指定年龄

3、返回值

# 计算并返回结果
def add(a, b):
    return a + b

result = add(3, 5)
print(result)  # 8

4、模块化：把代码分成多个文件

把常用函数放到单独的文件里：

# utils.py - 工具函数文件

def format_money(amount):
    """格式化金额"""
    return f'¥{amount:,.2f}'

def format_phone(phone):
    """格式化手机号"""
    return f'{phone[:3]}-{phone[3:7]}-{phone[7:]}'

def validate_email(email):
    """验证邮箱"""
    return '@' in email

然后在其他文件里使用：

# main.py - 主程序

from utils import format_money, format_phone

money = format_money(1234567.89)
print(money)  # ¥1,234,567.89

phone = format_phone('13812345678')
print(phone)  # 138-1234-5678

5、实战案例：模块化处理Excel

# excel_utils.py - Excel处理工具

import office

def read_sales(file_path):
    """读取销售数据"""
    return office.excel.read(path=file_path)

def calculate_total(df):
    """计算销售总额"""
    return df['销售额'].sum()

def generate_summary(df):
    """生成汇总数据"""
    return {
        'total': calculate_total(df),
        'count': len(df),
        'avg': df['销售额'].mean()
    }

# main.py - 主程序

from excel_utils import read_sales, generate_summary

df = read_sales('销售数据.xlsx')
summary = generate_summary(df)

print(f'总销售额：¥{summary["total"]:,.2f}')
print(f'订单数：{summary["count"]}')

6、实战案例：模块化邮件发送

# email_utils.py - 邮件工具

import office
from datetime import date

def send_daily_report(file_path, recipients):
    """发送日报"""
    today = date.today().strftime('%Y-%m-%d')
    
    for recipient in recipients:
        office.email.send(
            email=recipient,
            title=f'【日报】{today}',
            content=f'您好，{today}的报告请查收。',
            attachment=file_path
        )
    
    return len(recipients)

def send_reminder(message, recipients):
    """发送提醒"""
    for recipient in recipients:
        office.email.send(
            email=recipient,
            title='提醒',
            content=message
        )

# main.py - 主程序

from email_utils import send_daily_report

send_daily_report(
    file_path='日报.xlsx',
    recipients=['领导@example.com', '主管@example.com']
)

7、总结

模块化编程的好处：

  • ✅ 代码复用：写一次，用多次
  • ✅ 易于维护：修改一处，不影响其他
  • ✅ 结构清晰：每个文件功能分明
  • ✅ 团队协作：各人负责不同模块
