Nginx多项目配置

# 同一域名不同路径配置不同端口
server {
    listen 80;
    server_name example.com;
    
    location /api1/ {
        proxy_pass http://localhost:3000/;
    }
    
    location /api2/ {
        proxy_pass http://localhost:4000/;
    }
}
