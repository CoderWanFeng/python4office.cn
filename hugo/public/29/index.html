2、中级：带图片的PPT

import office

def create_product_ppt(product, image_path):
    """创建带图片的产品PPT"""
    filename = f"PPT_{product['name']}.pptx"
    
    # 封面
    office.ppt.create(path=filename, title=product['name'])
    
    # 带图片的介绍页
    office.ppt.add(
        path=filename,
        title=product['name'],
        content=f"💰 价格：{product['price']}",
        image=image_path
    )
    
    # 详情页
    office.ppt.add(
        path=filename,
        title='课程详情',
        content=product['desc']
    )
    
    return filename

# 批量创建
for product in products:
    # 假设产品图片是 product_图片.jpg
    img = f"{product['name']}_图片.jpg"
    if os.path.exists(img):
        create_product_ppt(product, img)

3、高级：从Excel读取数据批量生成

先准备一个 Excel 文件 产品列表.xlsx：

名称 价格 描述 图片
产品A 99元 这是产品A的描述 a.jpg
产品B 199元 这是产品B的描述 b.jpg

然后用代码批量生成：

import office
import os

def batch_create_product_ppts(excel_file):
    """从Excel批量生成产品PPT"""
    
    # 读取产品数据
    df = office.excel.read(path=excel_file)
    
    success = 0
    for idx, row in df.iterrows():
        name = row['名称']
        price = row['价格']
        desc = row['描述']
        image = row.get('图片', None)
        
        filename = f"产品介绍_{name}.pptx"
        
        try:
            # 创建PPT
            office.ppt.create(path=filename, title=name)

            # 详细参数页
            office.ppt.add(path=filename, title='详细信息', content=desc)
            
            # 联系方式页

                path=filename,

4、实战案例：批量生成公司介绍PPT

每个分公司都要做一个介绍 PPT？

import office

branches = [
    {
        'name': '北京分公司',
        'established': '2010年',
        'employees': '50人',
        'revenue': '5000万',
        'address': '北京市朝阳区XX大厦'
    },
    {
        'name': '上海分公司',
        'established': '2015年',
        'employees': '30人',
        'revenue': '3000万',
        'address': '上海市浦东新区XX大楼'
    }
]

for b in branches:
    filename = f"公司介绍_{b['name']}.pptx"
    
    office.ppt.create(path=filename, title=b['name'])
    
    # 公司概览
    office.ppt.add(
        path=filename,
        title=f"{b['name']} 概览",
        content=f"成立时间：{b['established']}\n员工人数：{b['employees']}\n年营业额：{b['revenue']}"
    )
    
    # 办公地址
    office.ppt.add(
        path=filename,
        title='联系我们',
        content=f"地址：{b['address']}\n电话：xxx-xxxx-xxxx"
    )
    
    print(f'已生成: {filename}')

5、常见问题

Q：生成的PPT打开格式乱了？

A：python-pptx 对复杂模板支持有限，建议生成后用 PowerPoint 微调。

Q：图片路径找不到？

A：确保图片文件存在，可以用 os.path.exists() 检查。

Q：Mac上能用吗？

A：可以，但部分功能需要 Microsoft PowerPoint。
