```python
import office

# 1行代码，提取mp3文件
office.video.video2mp3(path=r'C:\Users\Administrator\Desktop\10讲python自动化办公\09-从视频里提取音频\44.mp4',\
                       mp3_name='44',output_path=r'C:\Users\Administrator\Desktop\10讲python自动化办公\09-从视频里提取音频\MP3文件')
```
