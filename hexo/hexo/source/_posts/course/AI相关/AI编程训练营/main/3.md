## Qoder的AI功能深度体验

### 智能代码生成：比传统IDE更懂你

在Qoder中，你只需要用自然语言描述需求，AI就会生成完整的`python-office`代码：

**操作步骤：**
1. 新建`file_organizer.py`文件
2. 输入注释：
```python
# 使用python-office创建一个智能文件整理程序
# 要求：按文件类型自动分类，支持图片、文档、压缩包
# 显示整理进度和结果统计
```
3. 按下`Ctrl+I`召唤AI助手，选择"生成代码"

**Qoder自动生成：**
```python
import office
import os
from pathlib import Path

def smart_file_organizer(source_folder, target_folder):
    """
    智能文件整理程序
    """
    # 创建目标文件夹
    folders = ['图片', '文档', '压缩包', '其他']
    for folder in folders:
        Path(os.path.join(target_folder, folder)).mkdir(parents=True, exist_ok=True)
    
    # 文件类型映射
    file_types = {
        '图片': ['.jpg', '.jpeg', '.png', '.gif', '.bmp', '.webp'],
        '文档': ['.pdf', '.docx', '.xlsx', '.pptx', '.txt', '.md'],
        '压缩包': ['.zip', '.rar', '.7z', '.tar', '.gz']
    }
    
    processed_files = 0
    for file_path in Path(source_folder).iterdir():
        if file_path.is_file():
            suffix = file_path.suffix.lower()
            moved = False
            
            for category, extensions in file_types.items():
                if suffix in extensions:
                    target_path = Path(target_folder) / category / file_path.name
                    office.file.move_file(str(file_path), str(target_path))
                    print(f'已移动: {file_path.name} -> {category}')
                    processed_files += 1
                    moved = True
                    break
            
            if not moved:
                target_path = Path(target_folder) / '其他' / file_path.name
                office.file.move_file(str(file_path), str(target_path))
                print(f'已移动: {file_path.name} -> 其他')
                processed_files += 1
    
    print(f'\n整理完成！共处理 {processed_files} 个文件')

# 使用示例
if __name__ == "__main__":
    smart_file_organizer('./下载文件夹/', './整理后的文件/')
```
